10 OPEN 9,8,9,"TEST0000.BIN":REM FILE HANDLE 9, DEVICE 8
15 M=255:REM MAX BYTES TO PEEK AT
20 BINPUT#9,A$,M:REM LOAD M BYTES INTO STR A$ (255 MAX)
30 PRINT LEN(A$);"BYTES"
40 CLOSE 9:REM CLOSE FILE HANDLE 9
50 FOR I = 1 TO M
60 B$=MID$(A$,I,1):REM GET I-TH BYTE FROM A$ AS CHAR.
70 PRINT I,ASC(B$):REM CONVERT CHAR TO ASCII VALUE
80 NEXT I
